            # YOLO 라벨 형식으로 변환
            center_x = ((x_min + x_max) / 2) / w
            center_y = ((y_min + y_max) / 2) / h
            width = (x_max - x_min) / w
            height = (y_max - y_min) / h

            # 0~1 범위를 보장
            center_x = max(0, min(center_x, 1))
            center_y = max(0, min(center_y, 1))
            width = max(0, min(width, 1))
            height = max(0, min(height, 1))
